```python
class Solution:
    def searchInsert(self, nums: List[int], target: int) -> int:
        if not nums: return 0

        start = 0
        end = len(nums)-1
        while start <= end:
            mid = (start + end) //2
            if nums[mid] == target:
                return mid 
            elif nums[mid] > target:
                end = mid - 1
            else:
                start = mid + 1
            if start-1 == end:
                return end + 1

```
